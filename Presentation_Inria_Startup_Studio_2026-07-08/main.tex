\section{Projet \& Solution}
% Slide 3: Pitch & Projet
\begin{frame}{Pitch \& Projet Percolia}
    \begin{block}{\large Percolia en une phrase}
        \textbf{Percolia} transforme automatiquement des nuages de points 3D LiDAR bruts et bruités en jumeaux numériques précis grâce à une nouvelle génération d'algorithmes de \textbf{clustering d'ordre supérieur}.
    \end{block}
    
    \vspace{0.4cm}
    
    \textbf{Pourquoi le LiDAR 3D ?}
    \begin{itemize}
        \item \textbf{Explosion des données :} Capteurs LiDAR embarqués partout (drones, robots, véhicules autonomes, infrastructures).
        \item \textbf{Besoins métiers :} Segmenter finement les objets (arbres, poteaux, bâtiments, pièces) pour l'inspection et la simulation.
        \item \textbf{Le goulot d'étranglement :} Le nettoyage et la structuration manuels des données coûtent cher et prennent du temps.
    \end{itemize}
    
    \vspace{0.4cm}
    \begin{center}
        \textcolor{inria-rouge}{\textbf{Notre force :}} Une technologie \textbf{sans apprentissage} (\emph{training-free}), hautement précise et résistante au bruit.
        \\[3pt]
        \scriptsize \textit{Note : La technologie est également généralisable à d'autres structures complexes comme les logs cyber (voir Backup).}
    \end{center}
\end{frame}

% Slide 4: Solution Percolia
\begin{frame}{Solution Percolia : Rupture par l'ordre supérieur}
    \begin{center}
        \large Comment Percolia résout le problème de segmentation 3D ?
    \end{center}
    
    \vspace{0.3cm}
    
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Le concept de connectivité d'ordre supérieur :}
            \begin{itemize}
                \item \textbf{Graphes (actuel) :} Deux points sont connectés s'ils sont proches. Le bruit de mesure relie facilement deux amas disjoints.
                \item \textbf{Hypergraphes (Percolia) :} On n'étudie plus les liaisons par paires, mais les intersections multiples de boules (complexes géométriques).
                \item \textbf{Retard de percolation :} L'obligation d'avoir des connexions multipoints bloque la propagation du bruit et isole naturellement les vrais amas.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.46\textwidth}
            \textbf{Impacts concrets pour nos utilisateurs (ex: Marie) :}
            \begin{itemize}
                \item \textbf{Détourage parfait :} Les structures métalliques du bateau et les pièces d'assemblage sont séparées de façon nette, sans bavure due au bruit.
                \item \textbf{Diagnostic automatique :} L'analyse de conformité géométrique devient fiable à 100\% (pièce bien placée vs mal placée).
                \item \textbf{Gain de temps :} Suppression des étapes manuelles de nettoyage de nuages de points.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\section{Technologie}
% Slide 5: Technologie sous-jacente : HGP-Clusterer
\begin{frame}{Technologie sous-jacente : HGP-Clusterer}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Origine académique :}
            \begin{itemize}
                \item Brevet et technologie issus des travaux de thèse de \textbf{Louis \textsc{Hauseux}} à l'Inria Sophia Antipolis (équipes \textsc{Neo} et \textsc{Ayana}).
                \item Fondements théoriques rigoureux alliant la \textbf{géométrie algorithmique} et la \textbf{théorie de la percolation}.
            \end{itemize}
            
            \vspace{0.3cm}
            \textbf{Algorithme HGP-Clusterer :}
            \begin{itemize}
                \item Substitution des graphes par des complexes de \v{C}ech et des composantes connexes par des \textbf{$K$-polyèdres}.
                \item Extraction rapide des fusions clés (simplexes de Gabriel d'ordre supérieur) via la mosaïque de Delaunay d'ordre $K$.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \begin{center}
                \textbf{\orange Avantages compétitifs clés}
            \end{center}
            \begin{description}
                \item[Training-Free] Aucun besoin de données étiquetées pour l'entraînement.
                \item[Précision intrinsèque] Robustesse mathématiquement prouvée face au bruit.
                \item[Priors géométriques] Flexibilité pour injecter des contraintes physiques (ex: volume attendu de l'objet).
            \end{description}
            
            \vspace{0.4cm}
            \begin{block}{Publication clé}
                L. \textsc{Hauseux}, K. \textsc{Avrachenkov}, J. \textsc{Zerubia}, \emph{Generalization of single-linkage with higher-order interactions}, \textbf{Applied Network Science}, 2026.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Backup 1 : Généralisation à l'axe Logs Cyber}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{\bleu Le problème : Les angles morts de sécurité}
            \begin{itemize}
                \item \textbf{Exemple (Le Louvre - \textsc{Abdelhadi}) :} Des caméras de sécurité mal placées créent des angles morts facilitant l'intrusion.
                \item \textbf{En cybersécurité/fraude :} Le flux de logs est massif. Sans corrélation intelligente, des alertes isolées masquent des attaques complexes (angles morts).
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{\bleu L'approche Percolia :}
            \begin{itemize}
                \item Transformation des logs informatiques en \textbf{embeddings} adaptés.
                \item Application de HGP-Clusterer pour regrouper les événements autour de sessions, IPs et comptes.
                \item Regroupement des logs liés pour réduire la masse d'alertes à analyser et faire émerger les comportements atypiques.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Backup 2 : Mosaïque de Delaunay d'ordre $K$ \& Simplexes de Gabriel}
    \begin{itemize}
        \item \textbf{Mosaïque de Delaunay d'ordre $K$ :} Généralisation de la triangulation classique. Elle contient tous les simplexes nécessaires pour calculer la hiérarchie $K$-NN.
        \item \textbf{Simplexes de Gabriel :} Un simplexe $\sigma$ est de Gabriel si sa boule circonscrite ne contient aucun point intrus dans $\mathcal{X} \setminus \sigma$.
        \item \textbf{Théorème :} Les fusions dans la hiérarchie HGP s'effectuent nécessairement le long de simplexes de Gabriel.
        \item \textbf{Impact algorithmique :} Cela réduit drastiquement l'espace de recherche et rend l'algorithme HGP-Clusterer extrêmement rapide sur de grandes dimensions.
    \end{itemize}
\end{frame}

\begin{frame}{Backup 3 : Résultats en Segmentation Temporelle 4D (SemanticKITTI)}
    \begin{itemize}
        \item Suivi temporel d'instances sur des nuages de points 3D séquentiels.
        \item Combinaison de \textbf{HGP-Clusterer} avec du \textbf{Transport Optimal Déséquilibré} (\emph{Unbalanced Optimal Transport}).
        \item Solution complètement \textbf{sans apprentissage} (comparée à Geo-4D ou Alpine).
        \item Élimination totale du besoin de réentraînement lors des changements de domaine de capteurs.
    \end{itemize}
\end{frame}
