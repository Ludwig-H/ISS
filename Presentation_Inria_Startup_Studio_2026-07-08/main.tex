% Slide 3: Le standard de l'état de l'art
\begin{frame}{Le standard de l'état de l'art du clustering : (H)DBSCAN}
    \textbf{\bleu L'algorithme de clustering incontournable pour tous nos cas d'usage :}
    \begin{itemize}
        \item C'est la brique de base utilisée par tous les leaders du secteur pour segmenter et structurer les nuages de points :
        \begin{itemize}
            \item \textbf{Greehill} : Gyula \textsc{Fekete} (CTO) et \href{https://www.linkedin.com/in/benjamin-berta/?locale=hu}{Benjamin \textsc{Berta}} (Lead Machine Learning).
            \item \textbf{Hydro-Québec} : \href{https://www.linkedin.com/in/philippemassicotte/}{Philippe \textsc{Massicotte}} (Chercheur).
            \item \textbf{Exwayz} : \href{https://www.linkedin.com/in/hassan-bouchiba-434078202/}{Hassan \textsc{Bouchiba}} (Co-fondateur \& CTO).
        \end{itemize}
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{\bleu Pourquoi un tel succès ?}
    \begin{itemize}
        \item \textbf{Principe simple :} Un clustering fondé uniquement sur la densité locale des points.
        \item \textbf{Grande disponibilité :} Intégré par défaut dans toutes les bibliothèques logicielles standards, le rendant universel et immédiat à déployer.
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{\bleu La limite critique :}
    \begin{itemize}
        \item Face au bruit de mesure (poussières, occlusion), l'analyse purement binaire par paires de points fusionne à tort des structures disjointes (effet de chaînage).
    \end{itemize}
\end{frame}

\section{Technologie}
% Slide 5: Rupture technologique : HGP-Clusterer
\begin{frame}{Rupture technologique : HGP-Clusterer}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{\bleu Les deux piliers scientifiques :}
            \begin{itemize}
                \item \textbf{L'ordre supérieur :} la connectivité multi-points (graphes $\rightarrow$ hypergraphes) permet de mieux rendre compte de la densité locale.
                \item \textbf{La percolation (seuil critique différé) :} retardement théorique de la connectivité globale bloquant la propagation du bruit de mesure.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.46\textwidth}
            \textbf{\orange Avantages \& Optimisation}
            \begin{itemize}
                \item \textbf{Training-Free :} Aucun besoin de labellisation, robuste au changement de capteur.
                \item \textbf{Priors géométriques :} Possibilité d'injecter des \emph{a priori} géométriques (volumes, modèle 3D théorique).
                \item \textbf{Rapide :} l'algorithme a été optimisé pour passer à l'échelle industrielle (ex. : 10~000~000 de points lors de la collaboration avec Marie \textsc{Aspro}).
            \end{itemize}
            
            \vspace{0.25cm}
            \begin{block}{Publication de référence}

                L. \textsc{Hauseux}, K. \textsc{Avrachenkov}, J. \textsc{Zerubia}, \emph{Generalization of single-linkage with higher-order interactions}, \textbf{Applied Network Science}, 2026.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Clustering on Euclidean Data $\mathcal{X} \subset \mathbb{R}^d$: Density Estimator}
    \begin{center}
        \Large $\widehat{f} \leftarrow \widehat{f}_{K-\text{NN}}$ \\[4pt]
        \normalsize \textbf{Estimator}: $K$-Nearest Neighbors ($K$-NN)
    \end{center}
    
    \vspace{0.4cm}
    
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.56\textwidth}
            \begin{center}
                \vspace{0.2cm}
                {\Large\mathversion{bold} $\forall x \in \mathbb{R}^d, \quad \widehat{f}_{K-\text{NN}}(x) = \frac{K}{n \, \omega_d \, r_K(x)^d}$}
                
                \vspace{1.0cm}
                
                {\large\mathversion{bold} $r_K(x) = \min \left\{ r \ge 0 \;\middle|\; |B(x,r) \cap \mathcal{X}| \ge K \right\}$}
            \end{center}
        \end{column}
        
        \begin{column}{0.42\textwidth}
            \begin{itemize}
                \setlength{\itemsep}{8pt}
                \item \textbf{$n$}: number of points in $\mathcal{X}$
                \item \textbf{$\omega_d$}: volume of the unit ball in $\mathbb{R}^d$
                \item \textbf{$r_K(x)$}: distance to the $K$-th nearest neighbor of $x$
                \item \textbf{$\omega_d r_K(x)^d$}: volume of the ball containing the $K$ neighbors
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

    \begin{columns}[c, onlytextwidth]

            \vspace{0.3cm}

                \setlength{\itemsep}{10pt}

                \begin{itemize}

\begin{frame}{Backup 6 : Détection de Communautés sur Graphes Signés}
  \begin{block}{Détection de communautés sur graphes signés ($K=2$)}
    \begin{itemize}
      \item Partitionner les n\oe{}uds en $K=2$ communautés : $\Sigma_i \in \{+1, -1\}$.
      \item Les liens représentent des \textbf{attractions} (pleins) et des \textbf{répulsions} (pointillés).
      \item Au-delà du SBM classique : couplage entre géométrie spatiale et structure de communauté !
    \end{itemize}
  \end{block}
  \begin{block}{Le défi géométrique}
    \begin{itemize}
      \item Plongement spatial (ex: positions sur $\mathbb{Z}$ ou $\mathbb{R}^d$).
      \item Probabilités d'arêtes dépendant de la distance $r = \|x_i - x_j\|$ :
        \begin{align*}
            \mathbb{P}(\text{arête attractive} \mid \text{même communauté}) &= f_{\mathrm{in}}(r) \\
            \mathbb{P}(\text{arête attractive} \mid \text{communautés différentes}) &= f_{\mathrm{out}}(r)
        \end{align*}
      \item \textbf{Intérêt principal et verrou} : capturer cette géométrie sous-jacente.
    \end{itemize}
  \end{block}
\end{frame}
