% Slide 3: Problème adressé (Concrete and Segmented)
\begin{frame}{Problème adressé : Segments industriels \& Limites actuelles}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{\bleu 1. Segment LiDAR \& Jumeaux Numériques}
            \begin{itemize}
                \item \textbf{Cibles :} \textbf{Greehill} (monitoring vert), \textbf{Exwayz} (logiciels 3D), \textbf{YellowScan} (matériel).
                \item \textbf{Problème :} Les algorithmes classiques (DBSCAN, HDBSCAN) souffrent de \emph{l'effet de chaînage}. Le bruit relie à tort des objets séparés, obligeant à des corrections manuelles coûteuses.
                \item \textbf{Exemple (Montage bateau - Marie) :} Actuellement, le logiciel classe mal les pièces («~présentes~» vs «~absentes~», «~bien~» vs «~mal placées~»), ce qui bloque le montage.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{\bleu 2. Segment Logs \& Cybersécurité}
            \begin{itemize}
                \item \textbf{Cibles :} SOC (Security Operation Centers), analystes fraude, \textbf{Datadog} (échanges Vivatech).
                \item \textbf{Problème :} Des millions de logs bruts non structurés. Les outils actuels ratent les corrélations fines et génèrent trop de faux positifs.
                \item \textbf{Exemple (Le Louvre - Abdelhadi) :} Comme des caméras mal positionnées qui créent des angles morts, le manque de regroupement des logs laisse des failles de sécurité invisibles pour l'analyste.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

% Slide 4: Solution Percolia
\begin{frame}{Solution Percolia : L'apport de l'ordre supérieur}
    \begin{center}
        \large Comment Percolia répond à ces limites ?
    \end{center}
    
    \vspace{0.2cm}
    
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Une rupture conceptuelle :}
            \begin{itemize}
                \item \textbf{Graphes binaires (actuel) :} Connectent les points par de simples paires d'arêtes. Très sensibles au bruit.
                \item \textbf{Interactions d'ordre supérieur (Percolia) :} Regroupent les points à l'aide d'interactions multipoints (hypergraphes géométriques).
                \item \textbf{Blocage de la percolation :} Les hypergraphes empêchent le bruit local de se propager et de fusionner à tort des amas distincts.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.46\textwidth}
            \textbf{Impacts métiers concrets :}
            \begin{itemize}
                \item \textbf{Pour Marie (Bateaux) :} Les pièces sont segmentées de façon nette. Le diagnostic «~bien~» vs «~mal placé~» devient instantané et fiable.
                \item \textbf{Pour Abdelhadi (Sécurité/Louvre) :} Plus d'angles morts. Les logs de sessions, IPs et comptes sont corrélés en groupes cohérents, priorisant l'alerte sur l'anomalie réelle.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

% Slide 5: Technologie sous-jacente : HGP-Clusterer
\begin{frame}{Technologie sous-jacente : HGP-Clusterer}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{Origine académique :}
            \begin{itemize}
                \item Technologie brevetable issue des travaux de thèse de \textbf{Louis Hauseux} à l'Inria Sophia Antipolis (équipes \textsc{Neo} et \textsc{Ayana}).
                \item Fondements théoriques sur les \textbf{complexes simpliciaux} et la \textbf{théorie de la percolation}.
            \end{itemize}
            
            \vspace{0.3cm}
            \textbf{Algorithme HGP-Clusterer :}
            \begin{itemize}
                \item Remplace les graphes par des complexes de \v{C}ech et les composantes par des \textbf{$K$-polyèdres}.
                \item Identifie exactement les amas de forte densité pour tout $K \ge 2$.
                \item Exploite la mosaïque de Delaunay d'ordre supérieur et les simplexes de Gabriel pour des calculs ultra-rapides.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \begin{center}
                \textbf{\orange Avantages compétitifs clés}
            \end{center}
            \begin{description}
                \item[Training-Free] Pas besoin d'entraînement ou de labels coûteux.
                \item[Ultra-précis] Résistance inégalée au bruit de fond (pas de chaînage).
                \item[Priors géométriques] Injection facile d'a priori métiers (ex: contraintes de volume).
            \end{description}
            
            \vspace{0.4cm}
            \begin{block}{Publication de référence}
                L. Hauseux, K. Avrachenkov, J. Zerubia, \emph{Generalization of single-linkage with higher-order interactions}, \textbf{Applied Network Science}, 2026.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

% --- ANNEXES (TECHNICAL BACKUP) ---
\begin{frame}{Annexe 1 : Mosaïque de Delaunay d'ordre $K$ \& Simplexes de Gabriel}
    \begin{itemize}
        \item \textbf{Mosaïque de Delaunay d'ordre $K$ :} Généralisation de la triangulation classique. Elle contient tous les simplexes nécessaires pour calculer la hiérarchie $K$-NN.
        \item \textbf{Simplexes de Gabriel :} Un simplexe $\sigma$ est de Gabriel si sa boule circonscrite ne contient aucun point intrus dans $\mathcal{X} \setminus \sigma$.
        \item \textbf{Théorème :} Les simplexes à l'origine de fusions de clusters dans la hiérarchie HGP sont nécessairement des simplexes de Gabriel.
        \item \textbf{Impact algorithmique :} Cela réduit drastiquement l'espace de recherche et rend l'algorithme HGP-Clusterer extrêmement rapide sur de grandes dimensions.
    \end{itemize}
\end{frame}

\begin{frame}{Annexe 2 : Résultats en Segmentation Temporelle 4D (SemanticKITTI)}
    \begin{itemize}
        \item Suivi temporel d'instances sur des nuages de points 3D séquentiels.
        \item Combinaison de \textbf{HGP-Clusterer} avec du \textbf{Transport Optimal Déséquilibré} (\emph{Unbalanced Optimal Transport}).
        \item Solution complètement \textbf{sans apprentissage} (comparée à Geo-4D ou Alpine).
        \item Élimination totale du besoin de réentraînement lors des changements de domaine de capteurs.
    \end{itemize}
\end{frame}
