% Slide 3: Le standard de l'état de l'art
\begin{frame}{Le standard de l'état de l'art du clustering : (H)DBSCAN}
    \textbf{L'algorithme de clustering incontournable pour tous nos cas d'usage :}
    \begin{itemize}
        \item C'est la brique de base utilisée par tous les leaders du secteur pour segmenter et structurer les nuages de points :
        \begin{itemize}
            \item \textbf{Greehill} : Gyula \textsc{Fekete} (CTO) et \href{https://www.linkedin.com/in/benjamin-berta/?locale=hu}{Benjamin \textsc{Berta}} (Lead Machine Learning).
            \item \textbf{Hydro-Québec} : \href{https://www.linkedin.com/in/philippemassicotte/}{Philippe \textsc{Massicotte}} (Chercheur).
            \item \textbf{Exwayz} : \href{https://www.linkedin.com/in/hassan-bouchiba-434078202/}{Hassan \textsc{Bouchiba}} (Co-fondateur \& CTO).
        \end{itemize}
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Pourquoi un tel succès ?}
    \begin{itemize}
        \item \textbf{Principe simple :} Un clustering fondé uniquement sur la densité locale des points.
        \item \textbf{Grande disponibilité :} Intégré par défaut dans toutes les bibliothèques logicielles standards, le rendant universel et immédiat à déployer.
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{La limite critique :}
    \begin{itemize}
        \item Face au bruit de mesure (poussières, occlusion), l'analyse purement binaire par paires de points fusionne à tort des structures disjointes (effet de chaînage).
    \end{itemize}
\end{frame}

\section{Technologie}
% Slide 5: Rupture technologique : HGP-Clusterer
\begin{frame}{Rupture technologique : HGP-Clusterer}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Les deux piliers scientifiques :}
            \begin{itemize}
                \item \textbf{L'ordre supérieur :} connectivité multipoints (intersections de boules multiples) modélisant exactement l'estimateur de densité des $K$ plus proches voisins ($\widehat f_{K\text{-NN}}$).
                \item \textbf{La percolation (seuil critique différé) :} retardement théorique de la connectivité globale bloquant la propagation du bruit de mesure.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.46\textwidth}
            \textbf{\orange Avantages \& Optimisation}
            \begin{itemize}
                \item \textbf{Training-Free :} Aucun besoin de labellisation, robuste au changement de capteur.
                \item \textbf{Priors géométriques :} Possibilité d'injecter des \emph{a priori} géométriques (volumes, modèle 3D théorique).
                \item \textbf{Rapide :} Porté par la mosaïque de Delaunay d'ordre $K$.
            \end{itemize}
            
            \vspace{0.2cm}
            \begin{block}{Publication de référence}
                \scriptsize
                L. \textsc{Hauseux}, K. \textsc{Avrachenkov}, J. \textsc{Zerubia}, \emph{Generalization of single-linkage with higher-order interactions}, \textbf{Applied Network Science}, 2026.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Backup 1 : Généralisation à l'axe Logs Cyber}
    \begin{columns}[T]
        \begin{column}{0.48\textwidth}
            \textbf{\bleu Le problème : Les angles morts de sécurité}
            \begin{itemize}
                \item \textbf{Exemple (Le Louvre - \textsc{Abdelhadi}) :} Des caméras de sécurité mal placées créent des angles morts facilitant l'intrusion.
                \item \textbf{En cybersécurité/fraude :} Le flux de logs est massif. Sans corrélation intelligente, des alertes isolées masquent des attaques complexes (angles morts).
            \end{itemize}
        \end{column}
        
        \begin{column}{0.48\textwidth}
            \textbf{\bleu L'approche Percolia :}
            \begin{itemize}
                \item Transformation des logs informatiques en \textbf{embeddings} adaptés.
                \item Application de HGP-Clusterer pour regrouper les événements autour de sessions, IPs et comptes.
                \item Regroupement des logs liés pour réduire la masse d'alertes à analyser et faire émerger les comportements atypiques.
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Backup 2 : Mosaïque de Delaunay d'ordre $K$ \& Simplexes de Gabriel}
    \begin{itemize}
        \item \textbf{Mosaïque de Delaunay d'ordre $K$ :} Généralisation de la triangulation classique. Elle contient tous les simplexes nécessaires pour calculer la hiérarchie $K$-NN.
        \item \textbf{Simplexes de Gabriel :} Un simplexe $\sigma$ est de Gabriel si sa boule circonscrite ne contient aucun point intrus dans $\mathcal{X} \setminus \sigma$.
        \item \textbf{Théorème :} Les fusions dans la hiérarchie HGP s'effectuent nécessairement le long de simplexes de Gabriel.
        \item \textbf{Impact algorithmique :} Cela réduit drastiquement l'espace de recherche et rend l'algorithme HGP-Clusterer extrêmement rapide sur de grandes dimensions.
    \end{itemize}
\end{frame}

\begin{frame}{Backup 3 : Résultats en Segmentation Temporelle 4D (SemanticKITTI)}
    \begin{itemize}
        \item Suivi temporel d'instances sur des nuages de points 3D séquentiels.
        \item Combinaison de \textbf{HGP-Clusterer} avec du \textbf{Transport Optimal Déséquilibré} (\emph{Unbalanced Optimal Transport}).
        \item Solution complètement \textbf{sans apprentissage} (comparée à Geo-4D ou Alpine).
        \item Élimination totale du besoin de réentraînement lors des changements de domaine de capteurs.
    \end{itemize}
\end{frame}
