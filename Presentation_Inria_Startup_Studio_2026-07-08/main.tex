\section{Projet \& Solution}
% Slide 3: Le standard de l'état de l'art
\begin{frame}{Le standard de l'état de l'art du clustering : (H)DBSCAN}
    \textbf{L'algorithme de clustering incontournable pour tous nos cas d'usage :}
    \begin{itemize}
        \item C'est la brique de base utilisée par tous les leaders du secteur pour segmenter et structurer les nuages de points :
        \begin{itemize}
            \item \textbf{Greehill} : Gyula \textsc{Fekete} (CTO) et \href{https://www.linkedin.com/in/benjamin-berta/?locale=hu}{Benjamin \textsc{Berta}} (Lead Machine Learning).
            \item \textbf{Hydro-Québec} : \href{https://www.linkedin.com/in/philippemassicotte/}{Philippe \textsc{Massicotte}} (Chercheur).
            \item \textbf{Exwayz} : \href{https://www.linkedin.com/in/hassan-bouchiba-434078202/}{Hassan \textsc{Bouchiba}} (Co-fondateur \& CTO).
        \end{itemize}
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{Pourquoi un tel succès ?}
    \begin{itemize}
        \item \textbf{Principe simple :} Un clustering fondé uniquement sur la densité locale des points.
        \item \textbf{Grande disponibilité :} Intégré par défaut dans toutes les bibliothèques logicielles standards, le rendant universel et immédiat à déployer.
    \end{itemize}
    
    \vspace{0.3cm}
    
    \textbf{La limite critique :}
    \begin{itemize}
        \item Face au bruit de mesure (poussières, occlusion), l'analyse purement binaire par paires de points fusionne à tort des structures disjointes (effet de chaînage).
    \end{itemize}
\end{frame}

\section{Technologie}
% Slide 5: Rupture technologique : HGP-Clusterer
\begin{frame}{Rupture technologique : HGP-Clusterer}
    \begin{columns}[T]
        \begin{column}{0.5\textwidth}
            \textbf{Les deux piliers scientifiques :}
            \begin{itemize}
                \item \textbf{L'ordre supérieur :} connectivité multipoints (intersections de boules multiples) modélisant exactement l'estimateur de densité des $K$ plus proches voisins ($\widehat f_{K\text{-NN}}$).
                \item \textbf{La percolation (seuil critique différé) :} retardement théorique de la connectivité globale bloquant la propagation du bruit de mesure.
            \end{itemize}
        \end{column}
        
        \begin{column}{0.46\textwidth}
            \textbf{\orange Avantages \& Optimisation}
            \begin{itemize}
                \item \textbf{Training-Free :} Aucun besoin de labellisation, robuste au changement de capteur.
                \item \textbf{Priors géométriques :} Possibilité d'injecter des \emph{a priori} géométriques (volumes, modèle 3D théorique).
                \item \textbf{Rapide :} Porté par la mosaïque de Delaunay d'ordre $K$.
            \end{itemize}
            
            \vspace{0.2cm}
            \begin{block}{Publication de référence}
                \scriptsize
                L. \textsc{Hauseux}, K. \textsc{Avrachenkov}, J. \textsc{Zerubia}, \emph{Generalization of single-linkage with higher-order interactions}, \textbf{Applied Network Science}, 2026.
            \end{block}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{The Case $K=1$: The \textsc{Gabriel} Graph}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.54\textwidth}
            \begin{itemize}
                \setlength{\itemsep}{10pt}
                \item \textbf{Definition}:
                \begin{itemize}
                    \item An edge $[A, B]$ is a \textsc{Gabriel} edge if its \textbf{diametral disk} does not contain any other point in $\mathcal{X}$.
                \end{itemize}
                \item \textbf{Intermediate structure}:
                \begin{itemize}
                    \item Key structure between the MST and the triangulation:
                    \[ \text{MST} \ \subseteq \ \text{\textsc{Gabriel} Graph} \ \subseteq \ \text{\textsc{Delaunay}} \]
                \end{itemize}
            \end{itemize}
        \end{column}

\begin{frame}{Generalization to Order $K$: Splitting Simplices}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.54\textwidth}
            \begin{itemize}
                \setlength{\itemsep}{10pt}
                \item \textbf{$K$-splitting simplex}:
                \begin{itemize}
                    \item It is a $K$-simplex that has a \textbf{real influence} (causes clusters to merge) on the $K$-NN hierarchy.
                    \item \textbf{Case $K=1$}: $1$-splitting simplices are precisely the \textbf{edges of the MST}.
                \end{itemize}
                \item \textbf{\textsc{Gabriel} condition}:
                \begin{itemize}
                    \item \textbf{Theorem}: any $K$-splitting simplex is necessarily a \textbf{\textsc{Gabriel} simplex} ($\mathring{B}_\sigma \cap (\mathcal{X} \setminus \sigma) = \varnothing$).
                    \item \textbf{Computational domain}: the \textbf{higher-order \textsc{Delaunay} mosaic} \cite{osangEdelsbrunner2023algorithm}.
                \end{itemize}
            \end{itemize}

\begin{frame}{The $K$-Minimum Spanning Tree Analogy}
    \begin{columns}[c, onlytextwidth]
        \begin{column}{0.54\textwidth}
            Similar to $K=1$, the higher-order hierarchy relies on a sparse combinatorial structure:
            
            \vspace{0.3cm}
            \begin{itemize}
                \setlength{\itemsep}{10pt}
                \item \textbf{Percolation objects}:
                \begin{itemize}
                    \item It is no longer points, but \textbf{facets} ($(K-1)$-simplices) that merge.
                \end{itemize}
                \item \textbf{\textsc{Gabriel} simplices}:
                \begin{itemize}
                    \item The simplices causing the merges (splitting simplices) are \textbf{\textsc{Gabriel}} simplices.
                \end{itemize}
                \item \textbf{Computational domain}:
                \begin{itemize}
                    \item The search is restricted to the simplices of the \textbf{higher-order \textsc{Delaunay} mosaic} \cite{osangEdelsbrunner2023algorithm}.
                \end{itemize}
            \end{itemize}
        \end{column}
